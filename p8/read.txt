# 12) Extract useful links using reranked results only
useful_links = []
seen = set()

for item in results:
    url = None

    if isinstance(item, dict):
        url = item.get("url")
    elif hasattr(item, "url"):
        url = getattr(item, "url", None)

    if url and url not in seen:
        useful_links.append(url)
        seen.add(url)

    if len(useful_links) >= 5:
        break
